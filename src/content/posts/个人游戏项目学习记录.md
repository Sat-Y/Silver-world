## 关于主角人物的代码解读 ##
```
const JUMP_VELOCITY := -320.0
const RUN_SPEED := 200.0
const FLOOR_ACCELERATE := RUN_SPEED/0.2  
#物理学，v=at，所以a=v/t
#const 定义常量——有固定值 > 相当于int a=1
#var 定义变量——值会发生变化

enum State {
	IDLE,RUNNING,
	JUMP,FALL,ATTACK1,ATTACK2,ATTACK3,HURT,DIE,DEFEND,SHANBI,
}
#enum是枚举类型，定义一组命名常量，这里是主角人物状态定义，便于后续状态机的实现，与c语言中的用法相同

@export var can_combo := false
@export var direction := Direction.RIGHT:
	set(v):
		direction = v
		if not is_node_ready():
			await ready
		graphics.scale.x = direction

@onready var sprite_2d: Sprite2D = $Sprite2D
@onready var animation_player: AnimationPlayer = $AnimationPlayer
#这里的@onready就是为了获取动画播放的功能，实现角色各个状态的动画播放
#@export是使所引用变量可以直接在面板编辑，方便调试效果
#@onready用于获取子节点，进而可以实现子节点的功能

func _ready() -> void:
	if stats:
		stats.health = stats.max_health  # 假设Stats类有max_health属性
	stand(default_gravity, 0.01)
#func _ready() -> void是GDscript定义函数，返回变量类型为void，即无函数返回值，c语言定义函数为 void Print_hello(),python语言定义函数为 def Print_hello(),不用定义返回类型

match state:
		State.IDLE:
			move(default_gravity, delta)
			
		State.RUNNING:
			move(default_gravity, delta)
#match与c语言中的switch类似，相当于多个if——else

```
